Solar Hub — Historical Usage Importer
Step 1 — Create the Supabase table (first time only)

Run this SQL in your Supabase SQL editor (open editor ↗):

CREATE TABLE IF NOT EXISTS solar_agl_usage ( id bigserial PRIMARY KEY, start_time timestamptz NOT NULL, end_time timestamptz NOT NULL, type text NOT NULL, -- 'Solar' or 'Generalusage' kwh numeric(10,4) NOT NULL DEFAULT 0, quality text, created_at timestamptz DEFAULT now() ); CREATE INDEX IF NOT EXISTS idx_agl_start ON solar_agl_usage (start_time); CREATE INDEX IF NOT EXISTS idx_agl_type ON solar_agl_usage (type, start_time); -- Allow anonymous reads and inserts ALTER TABLE solar_agl_usage ENABLE ROW LEVEL SECURITY; CREATE POLICY "public read" ON solar_agl_usage FOR SELECT USING (true); CREATE POLICY "public insert" ON solar_agl_usage FOR INSERT WITH CHECK (true); CREATE POLICY "public delete" ON solar_agl_usage FOR DELETE USING (true);
